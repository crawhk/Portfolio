library(ggeffects)
library(ggplot2)
# Generate predicted values from model
pred <- ggpredict(model, terms = c("DTTotalC", "Scenario"))
ggplot(pred, aes(x = x, y = predicted, color = group)) +
  geom_line(size = 1) +
  labs(x = "Dark Triad", 
       y = "Predicted Lying", 
       color = "Scenario", 
       fill = "Scenario", 
       title = "DT–Lying Relationship Moderated by Scenario")
